SELECT a.id, a.projCode, B.proj_code, b.proj_title, b.proj_desc, b.est_start_date, b.est_end_date, b.act_start_date, b.act_end_date, b.longitude, b.latitude, b.location, b.status,
(SELECT sum(tpp.weight_progress) FROM tbl_proj_progress AS tpp WHERE tpp.projCode = a.projCode) as total_weight_progress,
b.deleted,
b.by_id,
b.updated_by,
b.created_at,
b.updated_at

FROM tbl_proj_progress AS a LEFT JOIN tbl_projectlist AS b ON a.projCode = b.proj_code
GROUP BY a.projCode
ORDER BY created_at DESC

SELECT a.id, a.proj_code, a.proj_title, a.proj_desc,
(SELECT SUM(tpp.weight_progress) FROM tbl_proj_progress AS tpp LEFT JOIN tbl_task AS tt ON tpp.taskCode = tt.taskCode WHERE tpp.projCode = a.proj_code AND tpp.deleted = 0 AND tt.deleted = 0) AS total_weight_progress,
a.est_start_date, a.est_end_date, a.act_start_date, a.act_end_date,
a.longitude, a.latitude, a.location, a.status, a.deleted,
a.by_id, a.updated_by, a.created_at, a.updated_at 

FROM tbl_projectlist AS a order by a.created_at desc
